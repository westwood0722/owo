str2에 str1의 문자열을 복사하시오.

### 1. 제어문 사용
```cpp
#include <iostream>

#pragma warning(disable : 4996)

using namespace std;

int main(void)
{
    char str1[27] = "programming is interesting";
    char str2[27] = "difficult";
    
    for (i = 0; str1[i] != '\0'; i++)
    {
        str2[i] = str1[i];
    }
    cout << "str2 : " << str2 << endl;

    return 0;
}
```

### 2. strcpy 사용
```cpp
#include <iostream>

#pragma warning(disable : 4996)

using namespace std;

int main(void)
{
    char str1[27] = "programming is interesting";
    char str2[27] = "difficult";

    cout << "str2: " << strcpy(str2, str1) << endl;

    return 0;
}
```
