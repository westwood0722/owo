제어문을 사용하여 실수 a, b를 입력받아 덧셈, 뺄셈, 곱셈, 나눗셈을 할 수 있는 프로그램을 작성하시오.

```cpp

#include <iostream>

using namespace std;

int main()
{
	double a, b;
	char c;

	cin >> a >> c >> b;

	if (c == '+')
	{
		cout << a << " + " << b << " = " <<  a + b << endl;
	}
	else if ( c == '-')
	{
		cout << a << " - " << b << " = " << a - b << endl;
	}
	else if ( c == '*')
	{
		cout << a << " * " << b << " = " << a * b << endl;
	}
	else  if ( c == '%')
	{
		cout << a << " % " << b << " = " << a / b << endl;
	}
	else
	{
		cout << " try again " << endl;
	}
}

```
