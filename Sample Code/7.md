### 1. 2차원 배열 +90도 회전

```cpp
#include <iostream>

using namespace std;

void rotate() {
    char arr1[5][5];
    char arr2[5][5];
    int i, j;
    char alphabet = 'A';

    for (i = 0; i < 5; i++) {
        for (j = 0; j < 5; j++) {
            arr1[i][j] = alphabet;
            alphabet++;
        }
    }

    for (i = 0; i < 5; i++) {
        for (j = 0; j < 5; j++) {
            cout << arr1[i][j];
        }
        cout << endl;
    }

    cout << "==================" << endl;

    for (i = 0; i < 5; i++) {
        for (j = 0; j < 5; j++) {
            arr2[j][4 - i] = arr1[i][j];
        }
    }

    for (i = 0; i < 5; i++) {
        for (j = 0; j < 5; j++) {
            cout << arr2[i][j];
        }
        cout << endl;
    }

}

int main(void) {
    rotate();

    return 0;
}
```
